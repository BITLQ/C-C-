//统计一个字符串中 ’ * ‘ 的个数，同时将所有的‘ * ’
//移动到字符串的前面，如果字符串中有空格，则删除所有的空格。 

#include<stdio.h>
#include<stdlib.h>
#include<string.h>

void Func(char* arr)
{
	char* cur = arr;
	char str[100] = {0};

	int count = 0;
	int i = 0;
	int j = 0;
	int size =0;

	int len = strlen(arr);
	size = len;
	for(i = 0; i<len; i++)
	{
		if(arr[i] == ' ')
			size--;

		else if(arr[i] == '*')
		{
			count++;
			str[j] = '*';
		    j++;
		}
	}

	while(*cur)
	{
		if(*cur != ' ' && *cur != '*')
		{
			str[size - j] = *cur;
			j--;
		}
		cur++;
	}

	for(j = 0; j < strlen(str); j++)
		printf("%c",str[j]);
	printf("\n");
	printf("%d\n",count);
}

int main()
{
	char arr[100] = {0};
	scanf("%s",&arr);

	Func(arr);
	system("pause");
	return 0;
}
