#include<iostream>

using namespace  std;

void Hanoi(int n,char a, char b, char c)
{
	if(n == 1)
	{
		printf("%c->%c\n",a,c );
		return;
	}

	Hanoi(n-1,a,c,b);
	printf("%c->%c\n",a,c);
	Hanoi(n-1,b,a,c);
}

void test(int n)
{
	Hanoi(n,'A','B','C');
}

int main()
{
	int n = 4;
	test(n);

	system("pause");
	return 0;
}
