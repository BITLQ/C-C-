#include<string>
#include<iostream>

using namespace std;

int main()
{
	string s1;
	string s2;
	cin>>s1>>s2;
	
	string s3;
	int i = s1.size() - 1;
	int j = s2.size() - 1;
	int count = 0;

	int flag = s1.size() >= s2.size() ? 1:0;


	while(i >= 0  ||  j >= 0)
	{
		int sum = 0;
		if(i >= 0 && j >= 0)
		{
			sum = (s1[i] - '0') + (s2[j] - '0');
			sum += count;
		}
		else if(i >= 0)
		{
			sum = (s1[i] -'0')+count;
		}
		else
		{
			sum = (s2[j] -'0')+count;
		}

		if(flag == 1)
		{
			count = sum/10;
			s1[i] = sum%10 + '0';
		}
		else
		{
			count = sum/10;
			s2[j] = sum%10 + '0';
		}

		i--;
		j--;
	}

		if(count > 0)
		{
			int tmp = count;
			while(tmp)
			{
				count = tmp%10;
				if(flag == 1)
					s1.insert(s1.begin(),count+'0');
				else
					s2.insert(s2.begin(),count);
				tmp /= 10;
			}
			count = 0;
		}

		if(flag == 1)
		{
			for(int i = 0; i < s1.size(); ++i)
			{
				cout<<s1[i];
			}
			cout<<endl;
		}
		else
		{
			for(int i = 0; i < s2.size(); ++i)
			{
				cout<<s2[i];
			}
			cout<<endl;
		}

	return 0;
}
